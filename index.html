    <script>

        const principalInput = document.getElementById('principalInput');

        const yearsSlider = document.getElementById('yearsSlider');

        const contributionSlider = document.getElementById('contributionSlider');

        const rateSlider = document.getElementById('rateSlider');


        const yearsVal = document.getElementById('yearsVal');

        const contributionVal = document.getElementById('contributionVal');

        const rateVal = document.getElementById('rateVal');


        const totalValueDisplay = document.getElementById('totalValue');

        const totalSpentDisplay = document.getElementById('totalSpent');

        const totalInterestDisplay = document.getElementById('totalInterest');

        const investedPercent = document.getElementById('investedPercent');

        const interestPercent = document.getElementById('interestPercent');


        const ctx = document.getElementById('wealthChart').getContext('2d');

        let chart;


        const formatCurrency = (val) => {

            return new Intl.NumberFormat('en-US', {

                style: 'currency',

                currency: 'USD',

                maximumFractionDigits: 0

            }).format(val);

        };


        function initChart() {

            chart = new Chart(ctx, {

                type: 'doughnut',

                data: {

                    labels: ['Invested', 'Interest'],

                    datasets: [{

                        data: [0, 0],

                        backgroundColor: ['rgba(255, 255, 255, 0.2)', '#ffffff'],

                        borderColor: '#56958C',

                        borderWidth: 6,

                        hoverOffset: 0,

                        borderRadius: 20,

                        spacing: 4

                    }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    cutout: '85%',

                    plugins: {

                        legend: { display: false },

                        tooltip: {

                            enabled: true,

                            backgroundColor: '#2F5E5B',

                            titleFont: { size: 14, weight: 'bold' },

                            bodyFont: { size: 13 },

                            padding: 12,

                            displayColors: false,

                            callbacks: {

                                label: (item) => {

                                    const dataset = item.dataset;

                                    const total = dataset.data.reduce((acc, current) => acc + current, 0);

                                    const currentValue = dataset.data[item.dataIndex];

                                    const percentage = total > 0 ? Math.round((currentValue / total) * 100) : 0;

                                    return ` ${item.label}: ${formatCurrency(currentValue)} (${percentage}%)`;

                                }

                            }

                        }

                    }

                }

            });

        }


        function update() {

            const P = parseFloat(principalInput.value) || 0;

            const t = parseInt(yearsSlider.value);

            const PMT = parseInt(contributionSlider.value);

            const r = parseFloat(rateSlider.value) / 100;

            const n = 12;


            yearsVal.innerText = `${t}y`;

            contributionVal.innerText = formatCurrency(PMT);

            rateVal.innerText = `${rateSlider.value}%`;


            const ratePerPeriod = r / n;

            const numPeriods = n * t;

            const compoundFactor = Math.pow(1 + ratePerPeriod, numPeriods);


            const futureValue = (P * compoundFactor) + (PMT * ((compoundFactor - 1) / ratePerPeriod));

            const totalInvested = P + (PMT * numPeriods);

            const totalInterest = futureValue - totalInvested;

            

            const invPct = futureValue > 0 ? Math.round((totalInvested / futureValue) * 100) : 0;

            const intPct = futureValue > 0 ? Math.round((totalInterest / futureValue) * 100) : 0;


            totalValueDisplay.innerText = formatCurrency(futureValue);

            totalSpentDisplay.innerText = formatCurrency(totalInvested);

            totalInterestDisplay.innerText = formatCurrency(totalInterest);

            

            investedPercent.innerText = `${invPct}% OF TOTAL`;

            interestPercent.innerText = `${intPct}% OF TOTAL`;


            chart.data.datasets[0].data = [totalInvested, totalInterest];

            chart.update();

        }


        [principalInput, yearsSlider, contributionSlider, rateSlider].forEach(el => {

            el.addEventListener('input', update);

        });


        initChart();

        update();
